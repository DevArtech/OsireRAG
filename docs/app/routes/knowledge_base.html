Module: knowledge_base.py (Router)

This module contains the FastAPI router for the knowledge base. The knowledge base is a module that allows users to create a knowledge base, add documents to it, and search it for relevant information.

212async def search(
213    project_name: str,
214    vectorstore_name: str,
215    model_name: str,
216    search_parameters: SearchParameters,
217) -> StreamingResponse:
218    """
219    Search the knowledge base.
220
221    Args:
222    - `project_name (str)`: The name of the project.
223    - `vectorstore_name (str)`: The name of the vectorstore.
224    - `model_name (str)`: The name of the model.
225
226    Returns:
227    - StreamingResponse: The top chunks found for the given query.
228
229    Usage:
230    - POST /knowledge-base/search/
231
232    Author: Adam Haile  
233    Date: 10/16/2024
234    """
235    try:
236        chunks = kb.search(
237            project_name, vectorstore_name, model_name, search_parameters
238        )
239
240        # Asynchronously encode the chunks in JSON format and stream the response
241        async def async_json_encoder(chunks: List[Tuple[Chunk, float]]):
242            yield b'{"response": ['
243            for i, chunk in enumerate(chunks):
244                if i < len(chunks) - 1:
245                    yield json.dumps([chunk[0].model_dump(), float(chunk[1])]).encode(
246                        "utf-8"
247                    ) + b","
248                else:
249                    yield json.dumps([chunk[0].model_dump(), float(chunk[1])]).encode(
250                        "utf-8"
251                    )
252            yield b"]}\n"
253
254        return StreamingResponse(
255            async_json_encoder(chunks), media_type="application/json"
256        )
257    except ValueError as e:
258        # Return a 404 response if the project, vectorstore, or model is not found
259        return JSONResponse(status_code=404, content={"detail": str(e)})
