app.routes.llm

Module: llm.py (Router)

This module contains the FastAPI router for the large language model. The large language model is a module that allows users to interact with the RosieRAG model and recieve LLM summarizations of the chunks.

Classes:

  • Prompt: A recieve contract for the prompt endpoint.
  • RAGPrompt: A recieve contract for the RAG prompt endpoint.

Functions:

  • craft_rag_prompt: Crafts a new RAG prompt for the model.
  • rag_prompt: Streams the RAG response to the user.
  • prompt: FastAPI endpoint for prompting the LLM.
  • rag: FastAPI endpoint for prompting the model using RAG.

Attributes:

  • router: The FastAPI router object.
  • kb: The KnowledgeBase object.

Author: Adam Haile
Date: 10/30/2024

111def craft_rag_prompt(prompt: RAGPrompt) -> Tuple[str, List[Tuple[Chunk, float]]]:
112    """
113    Crafts a new RAG prompt for the model.
114
115    Args:
116    - `prompt (RAGPrompt)`: The RAG prompt to craft.
117
118    Returns:
119    - str: The new RAG prompt.
120    - List[Tuple[Chunk, float]]: The list of documents and their scores.
121
122    Usage:
123    - `craft_rag_prompt(prompt)`
124
125    Author: Adam Haile  
126    Date: 10/30/2024
127    """
128
129    # Log the received RAG prompt and search for documents
130    logger.info(f"Received RAG prompt: {prompt.params.query}")
131    documents = kb.search(
132        project_name=prompt.project_name,
133        vectorstore_name=prompt.vectorstore_name,
134        model_name=prompt.model_name,
135        params=prompt.params,
136    )
137    logger.info(
138        f"Retrieved documents: {'\n\n'.join([doc[0].content for doc in documents])}"
139    )
140
141    # Craft the new RAG prompt
142    new_prompt = textwrap.dedent(
143        f"""Contextual Information is below:
144            ------------------------------------------
145            {"\n\n".join([doc[0].content for doc in documents])}
146            ------------------------------------------
147            Use the context information, and not prior knowledge, to answer the query prompted by the user.
148            """
149    )
150
151    # Apply special prompting if using local llama-cpp-python model
152    if get_settings().ENVIRONMENT == "local":
153        new_prompt += textwrap.dedent(
154            """
155            \nEnd your response with \"Query:\"
156            Query: What was the average stock price of Amazon in 2020?
157            Answer: The average stock price per share of Amazon in 2020 was $3,311.37.
158            """
159        )
160
161    # Append the query to the prompt
162    new_prompt += f"\n\nQuery: {prompt.params.query}"
163
164    # Apply special prompting if using local llama-cpp-python model
165    if get_settings().ENVIRONMENT == "local":
166        new_prompt += "\n" + "Answer:"
167
168    return (
169        new_prompt,
170        documents,
171    )

174def rag_prompt(prompt: RAGPrompt) -> Iterator[str]:
175    """
176    Streams the RAG response to the user.
177
178    Args:
179    - `prompt (RAGPrompt)`: The RAG prompt to stream.
180
181    Returns:
182    - Iterator[str]: The stream of the RAG response.
183
184    Usage:
185    - `rag_prompt(prompt)`
186
187    Author: Adam Haile  
188    Date: 10/30/2024
189    """
190
191    # Get the RAG prompt and documents
192    contextual_prompt, documents = craft_rag_prompt(prompt)
193
194    # Prompt the LLM and yield the results
195    response = ""
196    generator = llm.stream_prompt(
197        contextual_prompt, history=prompt.conversation_history
198    )
199    for item in generator:
200        yield item.replace('"', '\\"')
201        response += item.replace('"', '\\"')
202
203    logger.info(f"Model Response: {response}")
204
205    # ID the chunks and yield them
206    yield "<|C|>"
207    for doc in documents:
208        yield json.dumps(doc[0].model_dump())
209
210    return

213@router.post("/prompt/")
214async def prompt(prompt: Prompt) -> StreamingResponse:
215    """
216    FastAPI endpoint for prompting the LLM.
217
218    Args:
219    - `prompt (Prompt)`: The prompt to send to the model.
220
221    Returns:
222    - StreamingResponse: The stream of the LLM response.
223
224    Usage:
225    - POST /llm/prompt/
226
227    Author: Adam Haile  
228    Date: 10/30/2024
229    """
230
231    # Apply special prompting if using local llama-cpp-python model
232    if get_settings().ENVIRONMENT == "local":
233        query = f'End your response with "Query:" Query: {prompt.prompt} Answer:'
234    else:
235        query = prompt.prompt
236
237    if prompt.stream:
238        # Stream the response using JSON encoding
239        def json_encoder(query):
240            try:
241                generator = llm.stream_prompt(query)
242                current_item = next(generator)
243                yield current_item.lstrip().replace('"', '\\"')
244                current_item = next(generator)
245
246                while True:
247                    try:
248                        yield current_item.replace('"', '\\"')
249                        current_item = next(generator)
250                    except StopIteration:
251                        yield current_item.rstrip().replace('"', '\\"')
252                        break
253
254            except StopIteration:
255                pass
256
257        return StreamingResponse(json_encoder(query), media_type="text/event-stream")
258
259    # Statically prompt the LLM response
260    return llm.prompt(query)
