131    def add_chunks(
132        self, vectorstore: Union[str, FAISS], chunks: List[EmbeddedChunk]
133    ) -> List[str]:
134        """
135        Adds chunks with embeddings to the vectorstore.
136
137        Args:
138        - `vectorstore (Union[str, FAISS])`: The vectorstore or its path.
139        - `chunks (List[EmbeddedChunk])`: The chunks to add.
140
141        Returns:
142        - List[str]: A list of IDs for the added chunks.
143
144        Usage:
145        - `chunk_ids = manager.add_chunks(vectorstore, embedded_chunks)`
146
147        Author: Adam Haile  
148        Date: 10/13/2024
149        """
150        # Load the vectorstore if it's a path
151        if isinstance(vectorstore, str):
152            vectorstore = self.load_vectorstore(vectorstore)
153
154        # Initialize the IDs and get the embeddings
155        ids = [str(uuid.uuid4()) for _ in range(len(chunks))]
156        embeddings = np.array([chunk.embedding for chunk in chunks], dtype=np.float32)
157
158        # Get the start index for the new embeddings
159        start_idx = vectorstore.index.ntotal
160
161        # Add the embeddings to the index
162        vectorstore.index.add(embeddings)
163
164        # Add the chunks to the docstore
165        for i, chunk in enumerate(chunks):
166            doc_id = ids[i]
167            vectorstore.docstore[doc_id] = {
168                "content": chunk.content,
169                "index": chunk.index,
170                **chunk.metadata,
171                "embedding": chunk.embedding,
172            }
173            vectorstore.index_to_docstore_id[start_idx + i] = doc_id
174
175        return ids

217        """
218        Deletes chunks from the vectorstore by their IDs.
219
220        Args:
221        - `vectorstore (Union[str, FAISS])`: The vectorstore or its path.
222        - `ids (List[str])`: A list of chunk IDs.
223
224        Returns:
225        - None
226
227        Usage:
228        - `manager.delete_chunks(vectorstore, chunk_ids)`
229
230        Author: Adam Haile  
231        Date: 10/13/2024
232        """
233        # Load the vectorstore if it's a path
234        if isinstance(vectorstore, str):
235            vectorstore = self.load_vectorstore(vectorstore)
236
237        # Get the indices to delete
238        indices_to_delete = [
239            idx
240            for idx, doc_id in vectorstore.index_to_docstore_id.items()
241            if doc_id in ids

243
244        # Delete the chunks from the docstore and index
245        for doc_id in ids:
246            if doc_id in vectorstore.docstore:

248
249        for idx in indices_to_delete:
250            del vectorstore.index_to_docstore_id[idx]
251
252        # Rebuild the index if necessary
253        if indices_to_delete:
254            new_index = faiss.IndexFlatL2(vectorstore.index.d)
255            remaining_embeddings = [
256                vectorstore.docstore[vectorstore.index_to_docstore_id[idx]]["embedding"]

312        """
313        Saves the vectorstore to disk.
314
315        Args:
316        - `vectorstore (FAISS)`: The vectorstore to save.

318
319        Returns:
320        - None
321
322        Usage:
323        - `manager.save_vectorstore(vectorstore, "./path")`
324
325        Author: Adam Haile  
326        Date: 10/13/2024
327        """
328        # Create the directory if it doesn't exist
329        os.makedirs(path, exist_ok=True)
330
331        # Save the index and docstore to disk
332        faiss.write_index(vectorstore.index, f"{path}/index.faiss")
333        with open(f"{path}/index.pkl", "wb") as f:
334            pickle.dump(

342    def load_vectorstore(self, path: str) -> FAISS:
343        """
344        Loads a vectorstore from disk.
345
346        Args:
347        - `path (str)`: The directory path to load the vectorstore from.
348
349        Returns:
350        - FAISS: The loaded vectorstore.
351
352        Usage:
353        - `vectorstore = manager.load_vectorstore("./path")`
354
355        Author: Adam Haile  
356        Date: 10/13/2024
357        """
358        # Load the index and docstore from disk
359        index = faiss.read_index(f"{path}/index.faiss")
360        with open(f"{path}/index.pkl", "rb") as f:
361            data = pickle.load(f)
362            docstore = data["docstore"]
363            index_to_docstore_id = data["index_to_docstore_id"]
364
365        # Return the vectorstore
366        return FAISS(
367            embedding_function=self.embedding_function,
368            index=index,
369            docstore=docstore,
370            index_to_docstore_id=index_to_docstore_id,
371        )
