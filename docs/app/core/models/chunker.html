 88    splitter: ClassVar[HTMLHeaderTextSplitter] = HTMLHeaderTextSplitter(headers_to_split_on)

 95    def chunk_document(self, document: Document, n: int = 3, **kwargs) -> List[Chunk]:
 96        """
 97        Chunks a document into smaller pieces.
 98
 99        Args:
100        - `document (Document)`: The document to chunk.
101        - `n (int)`: The number of sentences to include in each chunk.
102
103        Returns:
104        - List[Chunk]: The chunks of the document.
105
106        Usage:
107        - `chunks = chunk_document(document, n=7)`
108
109        Author: Adam Haile  
110        Date: 10/9/2024
111        """
112
113        # Additional optional kwargs for splitting text
114        max_char_length = kwargs.get("max_length", 10000)
115        overlap = kwargs.get("overlap", 50)
116
117        # Instantiate the RecursiveCharacterTextSplitter
118        recursive_splitter = RecursiveCharacterTextSplitter(
119            chunk_size=max_char_length,
120            chunk_overlap=overlap,
121            length_function=len,
122            is_separator_regex=False,
123        )
124
125        # Set the metadata for the chunks
126        metadata = {"directory": os.path.abspath(document.directory).replace("\\", "/")}
127
128        # Validate the document type and split accordingly
129        if document.directory.endswith(".pdf") or document.directory.endswith(".txt"):
130            merged_chunks = []
131
132            # Split the document into chunks of n sentences
133            chunks = recursive_splitter.split_text(document.content)
134
135            for chunk in chunks:
136                # Split the chunk into sentences
137                sentences = [
138                    str(sentence).strip() for sentence in self.nlp(chunk).sents
139                ]
140
141                # Merge the sentences into chunks of n sentences
142                for i in range(0, len(sentences), n):
143                    merged_chunks.append(" ".join(sentences[i : i + n]))
144
145            return [
146                Chunk(content=chunk, index=i, metadata=metadata)
147                for i, chunk in enumerate(merged_chunks)
148            ]
149
150        if document.directory.endswith(".html"):
151            # Split the document into chunks of n sentences
152            splits = self.splitter.split_text(document.content)
153
154            # Get the content of each split
155            splits = [split.page_content for split in splits]
156
157            return [
158                Chunk(content=chunk, index=i, metadata=metadata)
159                for i, chunk in enumerate(splits)
160            ]
161
162        return []
