222    def add_documents(
223        self,
224        args: DocumentArgs,
225        documents: List[UploadFile],
226        upload: bool = True,
227    ) -> List[str]:
228        """
229        Adds documents to the knowledge base.
230
231        Args:
232        - `args (DocumentArgs)`: The document arguments.
233        - `documents (List[UploadFile])`: The list of documents to add.
234        - `upload (bool)`: Whether to upload the documents or use pre-added ones.
235
236        Returns:
237        - List[str]: The list of document IDs.
238
239        Raises:
240        - ValueError: If a document of the same name already exists.
241
242        Usage:
243        - `kbase.add_documents(args, documents)`
244
245        Author: Adam Haile  
246        Date: 10/16/2024
247        """
248        # Track the time it takes to add the documents
249        start_time = time.time()
250
251        # Validate the project
252        project_path, vs_path, _ = self._validate_project(
253            args.project_name, args.vectorstore_name, args.model_name
254        )
255
256        ids = []
257        tokenized_docs = []
258        new_tokenized_docs = []
259
260        # Attempt to load the existing BM25 model and documents
261        try:
262            existing_docs, _ = self.bm25.load_model(args.project_name, args.model_name)
263            tokenized_docs.extend(existing_docs)
264            logger.info("Existing BM25 model and documents loaded.")
265        except FileNotFoundError:
266            # If no existing model, initialize a new list
267            logger.info("No existing BM25 model found. Creating a new one.")
268
269        for document in documents:
270            # Save the document to the project directory
271            file_path = os.path.join(project_path, document.filename)
272            if upload:
273                # Check if a document of the same name already exists
274                if os.path.exists(file_path):
275                    raise ValueError(
276                        f"Document of same name already exists: {document.filename}"
277                    )
278
279                os.makedirs(os.path.dirname(file_path), exist_ok=True)
280                with open(file_path, "wb") as f:
281                    f.write(document.file.read())
282
283            # Load the vectorstore
284            logger.info(f"Processing document: {document.filename}")
285            vectorstore = self.vs_manager.load_vectorstore(vs_path)
286
287            # Convert the document to a Document object
288            document_obj = Document(directory=os.path.abspath(file_path))
289
290            # Chunk, embed, and tokenize the document
291            logger.info("Chunking document.")
292            document_chunks = self.chunker.chunk_document(document_obj, n=args.n)
293
294            logger.info("Embedding document.")
295            document_embeddings = self.embedder.embed_chunks(document_chunks)
296
297            logger.info("Tokenizing document.")
298            document_tokens = self.tokenizer.tokenize_documents(document_chunks)
299            new_tokenized_docs.extend(document_tokens)
300
301            # Add the document embeddings to the vectorstore
302            ids.extend(self.vs_manager.add_chunks(vectorstore, document_embeddings))
303            logger.info(f"{document.filename} preprocessed.")
304
305        # Combine existing and new tokenized documents
306        tokenized_docs.extend(new_tokenized_docs)
307
308        # Save the vectorstore and create the new/updated BM25 model
309        logger.info("Saving databases and models.")
310        self.vs_manager.save_vectorstore(vectorstore, vs_path)
311        self.bm25.create_model(args.project_name, args.model_name, tokenized_docs)
312
313        # Log the time it took to add the documents
314        logger.info(f"Documents added in {time.time() - start_time} seconds.")
315
316        return ids

361    def add_project(
362        self, project_name: str, vectorstore_name: str, model_name: str
363    ) -> None:
364        project_path, vs_path, _ = self._validate_project(
365            project_name, vectorstore_name, model_name
366        )
367
368        documents = os.listdir(project_path)
369        if not documents:
370            raise ValueError("No documents found in project.")
371
372        vectorstore = self.vs_manager.load_vectorstore(vs_path)
373
374        document_chunks = []
375        for document in documents:
376            document_path = os.path.abspath(project_path + "/" + document).replace(
377                "\\", "/"
378            )
379            if not os.path.isdir(document_path):
380                document = Document(directory=document_path)
381                document_chunks.extend(self.chunker.chunk_document(document))
382
383        document_embeddings = self.embedder.embed_chunks(document_chunks)
384        self.vs_manager.add_chunks(vectorstore, document_embeddings)
385        self.vs_manager.save_vectorstore(vectorstore, vs_path)

387    def search(
388        self,
389        project_name: str,
390        vectorstore_name: str,
391        model_name: str,
392        params: SearchParameters,
393    ) -> List[Tuple[Chunk, float]]:
394        """
395        Searches the knowledge base.
396
397        Args:
398        - `project_name (str)`: The project name.
399        - `vectorstore_name (str)`: The vectorstore name.
400        - `model_name (str)`: The model name.
401
402        Returns:
403        - List[Tuple[Chunk, float]]: The search results and the score of the Chunk.
404
405        Raises:
406        - ValueError: If the project does not exist.
407
408        Usage:
409        - `kbase.search("project", "vectorstore", "model", SearchParameters(query="query", n_results=10, filter={}, rerank=True))`
410
411        Author: Adam Haile  
412        Date: 10/16/2024
413        """
414
415        # Track the time it takes to search the knowledge base
416        start_time = time.time()
417
418        # Validate the project
419        _, vs_path, _ = self._validate_project(
420            project_name, vectorstore_name, model_name
421        )
422
423        # Load the vectorstore, token chunks, and BM25 model
424        vectorstore = self.vs_manager.load_vectorstore(vs_path)
425        token_chunks, bm25_model = self.bm25.load_model(project_name, model_name)
426
427        # Embed and tokenize the query
428        query_embeddings = self.embedder.embed_query(params.query)
429        query_tokens = self.tokenizer.tokenize_query(params.query)
430
431        # Establish the search parameters for the vectorstore
432        vs_params = VectorstoreSearchParameters(
433            embedded_query=query_embeddings, k=params.n_results, filter=params.filter
434        )
435
436        # Search the vectorstore and BM25 model
437        faiss_chunks = self.vs_manager.search(vectorstore, vs_params)
438        keyword_chunks = self.bm25.search(
439            query_tokens, bm25_model, token_chunks, k=params.n_results
440        )
441
442        # Rank the results using reciprocal rank fusion
443        chunks = self.rrf.ranks([faiss_chunks, keyword_chunks], n=params.n_results)
444
445        # Rerank the results if necessary
446        if params.rerank:
447            chunks = self.reranker.cross_encode_rerank(
448                params.query, [chunk for chunk, score in chunks]
449            )
450
451        # Log the time it took to search the knowledge base
452        logger.info(f"Search completed in {time.time() - start_time} seconds.")
453
454        return chunks
